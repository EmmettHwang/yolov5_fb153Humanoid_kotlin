Android PacketBuilder (Kotlin)
object PacketBuilder {
    const val PACKET_SIZE = 15

    fun build(motionIndex: Int): ByteArray {
        require(motionIndex in 0..255) { "모션 인덱스 범위 오류" }

        val pkt = ByteArray(15)
        pkt[0] = 0xFF.toByte(); pkt[1] = 0xFF.toByte()
        pkt[2] = 0x4C;          pkt[3] = 0x53
        pkt[8] = 0x30;          pkt[9] = 0x0C
        pkt[10] = 0x03;         pkt[11] = motionIndex.toByte()
        pkt[13] = 100.toByte()   // 속도

        // 체크섬: byte[6]~[13] 합산 & 0xFF
        var chk = 0
        for (i in 6..13) chk = (chk + pkt[i].toInt() and 0xFF) and 0xFF
        pkt[14] = chk.toByte()
        return pkt
    }
}

JoystickView 핵심 구현 (Kotlin)
class JoystickView(context: Context) : View(context) {

    var onJoystickMove: ((JoystickOutput) -> Unit)? = null

    override fun onTouchEvent(event: MotionEvent): Boolean {
        when (event.action) {
            MotionEvent.ACTION_DOWN,
            MotionEvent.ACTION_MOVE -> updateStick(event.x, event.y)
            MotionEvent.ACTION_UP   -> resetToCenter()  // 손 떼면 정지
        }
        return true
    }

    private fun calcDirection(dx: Float, dy: Float, power: Float): Direction {
        if (power < 0.15f) return Direction.STOP  // 데드존
        val angle = Math.toDegrees(atan2(dy.toDouble(), dx.toDouble()))
        return when {
            angle in -67.5..-22.5  -> Direction.FORWARD_RIGHT
            angle in -112.5..-67.5 -> Direction.FORWARD
            angle in -157.5..-112.5-> Direction.FORWARD_LEFT
            angle >= 157.5 || angle <= -157.5 -> Direction.LEFT
            angle in 112.5..157.5 -> Direction.BACKWARD_LEFT
            angle in 67.5..112.5  -> Direction.BACKWARD
            angle in 22.5..67.5   -> Direction.BACKWARD_RIGHT
            else                   -> Direction.RIGHT
        }
    }
}

AudioPlayer 클래스 (Kotlin)
class AudioPlayer(private val context: Context) {

    private var mediaPlayer: MediaPlayer? = null

    // 로컬 파일 재생
    fun playFile(filePath: String) {
        release()
        mediaPlayer = MediaPlayer().apply {
            setDataSource(filePath)
            prepare(); start()
        }
    }

    // URL 스트리밍 재생
    fun playUrl(url: String) {
        release()
        mediaPlayer = MediaPlayer().apply {
            setDataSource(context, Uri.parse(url))
            prepareAsync()
            setOnPreparedListener { it.start() }
        }
    }

    // 앱 내 Asset (기본 사운드)
    fun playAsset(assetName: String) {
        release()
        val afd = context.assets.openFd(assetName)
        mediaPlayer = MediaPlayer().apply {
            setDataSource(afd.fileDescriptor, afd.startOffset, afd.length)
            prepare(); start()
        }
    }

    fun release() { mediaPlayer?.release(); mediaPlayer = null }
}
